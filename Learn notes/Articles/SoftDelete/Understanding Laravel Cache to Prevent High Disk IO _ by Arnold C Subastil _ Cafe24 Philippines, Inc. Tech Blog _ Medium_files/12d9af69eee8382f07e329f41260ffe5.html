<?php
public function permanentDeleteSoftDeleted($id)
{
$note = Note::onlyTrashed()->find($id);
if (!is_null($note)) {
$note->forceDelete();
$response = $this->successfulMessage(200, 'Successfully deleted', true, 0, $note);
} else {
return response($response);
}
return response($response);
}
