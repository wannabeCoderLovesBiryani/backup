<?php
public function restore($id)
{
$note = Note::onlyTrashed()->find($id);
if (!is_null($note)) {
$note->restore();
$response = $this->successfulMessage(200, 'Successfully restored', true, $note->count(), $note);
} else {
return response($response);
}
return response($response);
}
