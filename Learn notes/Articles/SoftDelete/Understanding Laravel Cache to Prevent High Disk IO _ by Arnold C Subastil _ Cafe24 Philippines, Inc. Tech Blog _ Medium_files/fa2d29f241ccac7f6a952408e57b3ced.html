<?php
namespace App\Http\Controllers;
class NotesController extends Controller
{
private function notFoundMessage()
{
return [
'code' => 404,
'message' => 'Note not found',
'success' => false,
];
}
private function successfulMessage($code, $message, $status, $count, $payload)
{
return [
'code' => $code,
'message' => $message,
'success' => $status,
'count' => $count,
'data' => $payload,
];
}
}
